import {
  ArgsTable,
  Canvas,
  Description,
  Meta,
  Story,
} from "@storybook/addon-docs"

import VSvg from "~/components/VSvg/VSvg.vue"
import { WithSuspense } from "~~/.storybook/decorators/with-suspense"

export const spriteData = [
  {
    name: "images",
    symbols: [
      "brand",
      "logo",
      "logo-and-wordmark",
      "oops",
      "openverse-logo-text",
      "radiomark",
      "wordpress",
    ],
  },
]

<Meta title="Components/VSvg" component={VSvg} decorators={WithSuspense} />

export const svgImages = spriteData.find(
  (sprite) => sprite.name === "images"
).symbols

export const Template = (args) => ({
  template: `<VSvg v-bind="args" />`,
  components: { VSvg },
  setup() {
    return { args }
  },
})

# SVG images

To display the svg, pass the name of the svg file as a prop, and add `viewbox`
or `width`/`height` pair.

```html
<VSvg name="image-name" />
```
